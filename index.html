Q.7 - This time we are required to print the total marks of the students with marks greater than 50 after a grace of 15 marks has been added to those students who scored less than 50.

OUTPUT:-
233

Ans - let studentRecords = [
    {name: 'John', id: 123, marks : 98 },
    {name: 'Baba', id: 101, marks : 23 },
    {name: 'yaga', id: 200, marks : 45 },
    {name: 'Wick', id: 115, marks : 75 }
  ];
  
  let updated= studentRecords.map(student => {
    if (student.marks < 50) {
      return { ...student, marks: student.marks + 15 };
    } else {
      return student;
    }
  });
  
  let filteredRecords = updated.filter(record => record.marks > 50);
  let totalMarks = filteredRecords.reduce((pre,cur) => pre+cur.marks, 0);
  console.log(totalMarks);
